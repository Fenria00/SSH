CREATE TABLE CFG_SYSTEM_CONFIG (
ID INT(18) NOT NULL COMMENT '系統參數編號',
CODE_CATE NVARCHAR(100) NOT NULL COMMENT '參數類別',
CATE_NAME NVARCHAR(200) NOT NULL COMMENT '類別名稱',
CODE NVARCHAR(100) NOT NULL COMMENT '代碼',
CODE_NAME NVARCHAR(500) NOT NULL COMMENT '代碼名稱',
CODE_VALUE NVARCHAR(500) NOT NULL COMMENT '代碼值',
CODE_DESC NVARCHAR(2000) COMMENT '代碼描述',
PARENT_ID INT(18) COMMENT '父代碼',
SEQ	INT(18) NOT NULL COMMENT '順序',
CREATE_DATE TIMESTAMP  NOT NULL DEFAULT NOW() COMMENT '建立日期',
CREATE_USER NVARCHAR(20) NOT NULL DEFAULT 'SYSTEM' COMMENT '建立人員',
UPDATE_DATE TIMESTAMP  NOT NULL DEFAULT NOW() COMMENT '更新日期',
UPDATE_USER NVARCHAR(20) NOT NULL DEFAULT 'SYSTEM' COMMENT '更新人員',
PRIMARY KEY(ID, CODE) 
)
