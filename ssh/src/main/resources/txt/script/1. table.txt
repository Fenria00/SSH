CREATE TABLE CFG_SYSTEM_CONFIG (
	ID INT(18) NOT NULL,
	CODE_CATE NVARCHAR(100) NOT NULL,
	CATE_NAME NVARCHAR(200) NOT NULL,
	CODE NVARCHAR(100) NOT NULL,
	CODE_NAME NVARCHAR(500) NOT NULL,
	CODE_VALUE NVARCHAR(500) NOT NULL,
	CODE_DESC NVARCHAR(2000),
	PARENT_ID INT(18) NOT NULL ,
	SEQ	INT(18) NOT NULL',
	CREATE_DATE TIMESTAMP NOT NULL DEFAULT NOW(),
	CREATE_USER NVARCHAR(20) NOT NULL DEFAULT 'SYSTEM',
	UPDATA_DATE TIMESTAMP  NOT NULL DEFAULT NOW(),
	UPDATA_USER NVARCHAR(20) NOT NULL DEFAULT 'SYSTEM',
	PRIMARY KEY(ID, CODE) 
)
